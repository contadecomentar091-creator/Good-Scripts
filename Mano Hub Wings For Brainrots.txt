local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

local Window = Rayfield:CreateWindow({
   Name = "Mano Hub",
   LoadingTitle = "Carregando Interface...",
   LoadingSubtitle = "by Gemini",
   ConfigurationSaving = {
      Enabled = false
   },
   Discord = {
      Enabled = false
   },
   KeySystem = false
})

-- Criando a Tab principal
local Tab = Window:CreateTab("Upgrades", 4483362458) -- Ícone padrão

-- Referências dos caminhos (facilitando o código)
local player = game:GetService("Players").LocalPlayer
local upgrades = player:WaitForChild("Upgrades")

local staminaLevel = upgrades:WaitForChild("StaminaLevel")
local speedLevel = upgrades:WaitForChild("SpeedLevel")
local carryLevel = upgrades:WaitForChild("CarryLevel")

-- Seção Visual
local Section = Tab:CreateSection("Alterar Atributos")

-- Input para Stamina Level
local InputStamina = Tab:CreateInput({
   Name = "Stamina Level",
   PlaceholderText = "Digite o valor atual: " .. tostring(staminaLevel.Value),
   RemoveTextAfterFocusLost = false,
   Callback = function(Text)
      local num = tonumber(Text)
      if num then
         staminaLevel.Value = num
         Rayfield:Notify({
            Title = "Sucesso!",
            Content = "Stamina Level alterado para " .. num,
            Duration = 3,
            Image = 4483362458,
         })
      else
         Rayfield:Notify({
            Title = "Erro",
            Content = "Por favor, insira apenas números.",
            Duration = 3,
            Image = 4483362458,
         })
      end
   end,
})
