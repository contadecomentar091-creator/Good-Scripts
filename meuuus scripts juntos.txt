local function startTpWalk()
    if loopRunning then return end
    loopRunning = true
    task.spawn(function()
        while tpwalking do
            local delta = hb:Wait()
            local chr = speaker.Character
            local hum = chr and chr:FindFirstChildWhichIsA("Humanoid")
            if chr and hum and hum.Parent then
                if hum.MoveDirection.Magnitude > 0 then
                    chr:TranslateBy(hum.MoveDirection * tpSpeed * delta * 10)
                end
            else
                tpwalking = false
                if TpToggleMain then TpToggleMain:Set(false) end
                if TpToggleStronguest then TpToggleStronguest:Set(false) end
            end
        end
        loopRunning = false
    end)
end

local function updateTpWalkState(isEnabled)
    tpwalking = isEnabled
    if isEnabled then startTpWalk() end
end

local function getLethalCharacter() return localPlayer.Character or localPlayer.CharacterAdded:Wait() end

local function alignCharacter()
    local chr = getLethalCharacter()
    local rPart = chr:WaitForChild("HumanoidRootPart")
    local hum = chr:WaitForChild("Humanoid")
    hum.AutoRotate = false
    local targetCFrame = rPart.CFrame * CFrame.Angles(0, math.rad(180), 0)
    rPart.CFrame = targetCFrame
    local magnitude = (camera.CFrame.Position - targetCFrame.Position).Magnitude
    camera.CFrame = CFrame.new(targetCFrame.Position - targetCFrame.LookVector * magnitude + Vector3.new(0, 2), targetCFrame.Position)
end

local function applyLaunch()
    local chr = getLethalCharacter()
    chr:WaitForChild("HumanoidRootPart").AssemblyLinearVelocity = Vector3.new(0, 64, 0)
end

local function stopLethalScript()
    if lethalAnimationConnection then lethalAnimationConnection:Disconnect() lethalAnimationConnection = nil end
end

local function startLethalScript()
    stopLethalScript()
    local chr = getLethalCharacter()
    local hum = chr:WaitForChild("Humanoid")
    lethalAnimationConnection = hum.AnimationPlayed:Connect(function(track)
        if not isLethalEnabled then return end
        if track.Animation.AnimationId == LETHAL_ANIMATION_ID then
            task.wait(1.7)
            applyLaunch()
            alignCharacter()
            local args = {{Dash = Enum.KeyCode.W, Key = Enum.KeyCode.Q, Goal = "KeyPress"}}
            chr:WaitForChild("Communicate"):FireServer(unpack(args))
            task.wait(LETHAL_DASH_DELAY)
            alignCharacter()
        end
    end)
end

local function playVisualSequence(duration)
	isProcessingSequence = true
	local characterHighlight = Instance.new("Highlight")
	characterHighlight.Parent = character
	characterHighlight.FillTransparency = 0.3
	characterHighlight.OutlineTransparency = 0.1
	local rainbowColors = {Color3.fromRGB(255, 0, 0), Color3.fromRGB(255, 127, 0), Color3.fromRGB(255, 255, 0), Color3.fromRGB(0, 255, 0), Color3.fromRGB(0, 0, 255), Color3.fromRGB(75, 0, 130), Color3.fromRGB(148, 0, 211)}
	local isRainbowActive = true
	task.spawn(function()
		local colorIndex = 1
		while isRainbowActive and characterHighlight.Parent do
			characterHighlight.FillColor = rainbowColors[colorIndex]
			characterHighlight.OutlineColor = rainbowColors[colorIndex % #rainbowColors + 1]
			colorIndex = colorIndex % #rainbowColors + 1
			task.wait(0.2)
		end
	end)
	local particleEmitter = Instance.new("ParticleEmitter")
	particleEmitter.Texture = "rbxassetid://243660364"
	particleEmitter.Rate = 6
	particleEmitter.Lifetime = NumberRange.new(1.2, 1.6)
	particleEmitter.Speed = NumberRange.new(0, 0)
	particleEmitter.Rotation = NumberRange.new(0, 360)
	particleEmitter.Size = NumberSequence.new({ NumberSequenceKeypoint.new(0, 2), NumberSequenceKeypoint.new(0.5, 6), NumberSequenceKeypoint.new(1, 10) })
	particleEmitter.Transparency = NumberSequence.new({ NumberSequenceKeypoint.new(0, 0.3), NumberSequenceKeypoint.new(1, 1) })
	particleEmitter.Color = ColorSequence.new({ ColorSequenceKeypoint.new(0, Color3.fromRGB(255, 255, 255)), ColorSequenceKeypoint.new(1, Color3.fromRGB(0, 255, 255)) })
	particleEmitter.LightEmission = 0.6
	particleEmitter.Parent = rootPart
	task.delay(duration, function()
		isProcessingSequence = false
		isRainbowActive = false
		if characterHighlight then characterHighlight:Destroy() end
		if particleEmitter then particleEmitter:Destroy() end
	end)
end

local function clearPhysicsFollow()
	if alignPosition then alignPosition:Destroy() end
	if alignOrientation then alignOrientation:Destroy() end
	if renderConnection then renderConnection:Disconnect() end
	if followPart then followPart:Destroy() end
	for _, child in ipairs(rootPart:GetChildren()) do
		if child:IsA("Attachment") or child.Name == "HasSnapped" then child:Destroy() end
	end
end

local function establishPhysicsFollow(targetRoot)
	clearPhysicsFollow()
	followPart = Instance.new("Part")
	followPart.Size = Vector3.new(0.5, 0.5, 0.5)
	followPart.Transparency = 1
	followPart.Anchored = true
	followPart.CanCollide = false
	followPart.Name = "FollowPart"
	followPart.Parent = workspace
	local att0_Pos = Instance.new("Attachment", rootPart)
	local att1_Pos = Instance.new("Attachment", followPart)
	local att0_Rot = Instance.new("Attachment", rootPart)
	local att1_Rot = Instance.new("Attachment", followPart)
	alignPosition = Instance.new("AlignPosition")
	alignPosition.Attachment0 = att0_Pos
	alignPosition.Attachment1 = att1_Pos
	alignPosition.RigidityEnabled = true
	alignPosition.Responsiveness = 200
	alignPosition.MaxForce = math.huge
	alignPosition.Parent = rootPart
	alignOrientation = Instance.new("AlignOrientation")
	alignOrientation.Attachment0 = att0_Rot
	alignOrientation.Attachment1 = att1_Rot
	alignOrientation.RigidityEnabled = true
	alignOrientation.Responsiveness = 100
	alignOrientation.MaxTorque = math.huge
	alignOrientation.Parent = rootPart
	renderConnection = RunService.RenderStepped:Connect(function()
		if targetRoot and targetRoot.Parent then
			local calculatedCFrame = targetRoot.CFrame * CFrame.new(0, 0, 0) * CFrame.Angles(math.rad(55), 0, 0)
			followPart.CFrame = calculatedCFrame
			if not rootPart:FindFirstChild("HasSnapped") then
				rootPart.CFrame = calculatedCFrame
				local tag = Instance.new("BoolValue")
				tag.Name = "HasSnapped"
				tag.Parent = rootPart
			end
		end
	end)
end

local function getClosestTarget()
	local closestDistance = maxTargetDistance
	local targetedRoot = nil
	local liveFolder = workspace:FindFirstChild("Live")
	if not liveFolder then return nil end
	for _, characterModel in ipairs(liveFolder:GetChildren()) do
		if characterModel:IsA("Model") and characterModel ~= character then
			local targetRoot = characterModel:FindFirstChild("HumanoidRootPart")
			local targetHumanoid = characterModel:FindFirstChildOfClass("Humanoid")
			if targetRoot and targetHumanoid and targetHumanoid.Health > 0 then
				local distance = (targetRoot.Position - rootPart.Position).Magnitude
				if distance <= closestDistance then
					if characterModel.Name == "Weakest Dummy" or Players:GetPlayerFromCharacter(characterModel) then
						targetedRoot = targetRoot
						closestDistance = distance
					end
				end
			end
		end
	end
	return targetedRoot
end

local function onAnimationPlayed(animationTrack)
	if not isScriptEnabled or isProcessingSequence then return end
	local animationId = string.match(animationTrack.Animation.AnimationId, "%d+")
	if BURST_ANIMATIONS[animationId] then
		isProcessingSequence = true
		task.delay(0.8, function() playVisualSequence(4.6) end)
	elseif DASH_ANIMATIONS[animationId] then
		isProcessingSequence = true
		task.delay(0.32, function()
			local communicationPayload = {{["Dash"] = Enum.KeyCode.W, ["Key"] = Enum.KeyCode.Q, ["Goal"] = "KeyPress"}}
			pcall(function() character:WaitForChild("Communicate"):FireServer(unpack(communicationPayload)) end)
			local priorityTarget = getClosestTarget()
			if priorityTarget then
				establishPhysicsFollow(priorityTarget)
				task.delay(0.5, clearPhysicsFollow)
			end
		end)
		task.delay(0.8, function() playVisualSequence(4.6) end)
	end
end

local function isAliveModel(model, exceptChar)
    if model == exceptChar then return false end
    local root = model:FindFirstChild("HumanoidRootPart")
    local humanoidObj = model:FindFirstChild("Humanoid")
    return root and humanoidObj and humanoidObj.Health > 0
end

local function findNearestKokusenTarget()
    local hrp = getJujutsuHRP()
    if not hrp then return nil end
    local myChar = localPlayer.Character
    local nearest = nil
    local bestDist = math.huge
    local function checkModel(model)
        if not isAliveModel(model, myChar) then return end
        local root = model:FindFirstChild("HumanoidRootPart")
        local dist = (hrp.Position - root.Position).Magnitude
        if dist < bestDist then
            bestDist = dist
            nearest = model
        end
    end
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= localPlayer and player.Character then checkModel(player.Character) end
    end
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("Model") then checkModel(obj) end
    end
    return nearest
end

local function curvedDashBehind(targetRoot, duration, customOffset)
    local char = localPlayer.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hrp or not hum or not targetRoot then return end
    hum.AutoRotate = false
    local startTime = tick()
    local startPos = hrp.Position
    local offsetDistance = customOffset or KOKUSEN_CONFIG.BehindOffset
    local conn
    conn = RunService.RenderStepped:Connect(function()
        if not targetRoot or not targetRoot.Parent then
            hum.AutoRotate = true
            conn:Disconnect()
            return
        end
        local elapsed = tick() - startTime
        local alpha = math.clamp(elapsed / duration, 0, 1)
        if alpha >= 1 then
            hum.AutoRotate = true
            conn:Disconnect()
            return
        end
        local smoothAlpha = 1 - (1 - alpha)^3
        local enemyCF = targetRoot.CFrame
        local enemyPos = enemyCF.Position
        local behindPos = enemyPos - (enemyCF.LookVector * offsetDistance)
        local basePos = startPos:Lerp(behindPos, smoothAlpha)
        local curve = enemyCF.RightVector * math.sin(smoothAlpha * math.pi) * offsetDistance
        local finalPos = basePos + curve
        hrp.CFrame = CFrame.lookAt(finalPos, enemyPos)
    end)
end

local targetRemote
pcall(function()
    targetRemote = ReplicatedStorage:WaitForChild("Knit",5):WaitForChild("Knit",5):WaitForChild("Services",5):WaitForChild("DivergentFistService",5):WaitForChild("RE",5):WaitForChild("Activated",5)
end)

if targetRemote then
    local oldNamecall
    oldNamecall = hookmetamethod(game, "__namecall", function(self, ...)
        if getnamecallmethod() ~= "FireServer" or self ~= targetRemote then return oldNamecall(self, ...) end
        if not autoKokusenEnabled or isCooling then return oldNamecall(self, ...) end
        isCooling = true
        local args = {...}
        local result = oldNamecall(self, ...)
        task.delay(KOKUSEN_CONFIG.FireDelay, function()
            pcall(function() targetRemote:FireServer(table.unpack(args)) end)
            isCooling = false
        end)
        task.delay(KOKUSEN_CONFIG.WarpDelay, function()
            local target = findNearestKokusenTarget()
            if not target then return end
            local targetRoot = target:FindFirstChild("HumanoidRootPart")
            if not targetRoot then return end
            local myHrp = getJujutsuHRP()
            local chosenOffset = KOKUSEN_CONFIG.BehindOffset
            if myHrp then
                local enemyLook = targetRoot.CFrame.LookVector
                local myLook = myHrp.CFrame.LookVector
                local dotProduct = enemyLook:Dot(myLook)
                if dotProduct > 0.5 then return end
                if math.abs(dotProduct) <= 0.5 then chosenOffset = KOKUSEN_CONFIG.BehindOffset * 0.5 end
            end
            curvedDashBehind(targetRoot, KOKUSEN_CONFIG.FireDelay, chosenOffset)
        end)
        return result
    end)
end

local function doYutaKillBlackflash()
    if isExecutingYuta or not localPlayer.Character then return end
    local char = localPlayer.Character
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local head = char:FindFirstChild("Head")
    local hum = char:FindFirstChildOfClass('Humanoid')
    if not hrp or not head or not hum then return end
    
    local targetPlayer = Players:FindFirstChild(selectedTargetName) -- Corrigido para ler a variavel certa
    local targetHrp = targetPlayer and targetPlayer.Character and targetPlayer.Character:FindFirstChild("HumanoidRootPart")
    local originalPos = hrp.CFrame
    
    isExecutingYuta = true
    hum.PlatformStand = true
    head.Anchored = true
    
    local moveset = char:FindFirstChild("Moveset")
    local move = moveset and moveset:FindFirstChild("Resolute Slash")
    
    if move then
        pcall(function() replicatesignal(localPlayer.Kill) end) -- Retornado conforme original
        task.wait(0.1)
        
        local target = nil
        pcall(function()
            local Knit = require(ReplicatedStorage.Knit.Knit)
            target = Knit.GetController("ToolController"):GetTarget()
        end)
        
        local skillTarget = (target and target.Parent) and target or char
        
        local remote = ReplicatedStorage:FindFirstChild("Knit")
        if remote then
            remote.Knit.Services.ResoluteSlashService.RE.Activated:FireServer(move, skillTarget)
        end
    end
    
    if CFloopYuta then CFloopYuta:Disconnect() end
    CFloopYuta = RunService.Heartbeat:Connect(function(deltaTime)
        if not isExecutingYuta or not head or not head.Parent then return end
        
        local camLook = camera.CFrame.LookVector
        local flatLook = Vector3.new(camLook.X, 0, camLook.Z).Unit
        
        if targetHrp and targetHrp.Parent then
            head.CFrame = CFrame.lookAt(targetHrp.Position + Vector3.new(0, 1.5, 1), targetHrp.Position + flatLook)
        else
            local moveDir = hum.MoveDirection
            if moveDir.Magnitude > 0 then
                moveDir = Vector3.new(moveDir.X, 0, moveDir.Z).Unit
                local newPos = head.Position + (moveDir * 28 * deltaTime)
                head.CFrame = CFrame.lookAt(newPos, newPos + flatLook)
            else
                head.CFrame = CFrame.lookAt(head.Position, head.Position + flatLook)
            end
        end
    end)
    
    task.delay(4, function()
        if CFloopYuta then CFloopYuta:Disconnect() CFloopYuta = nil end
        
        isExecutingYuta = false
        
        if head then head.Anchored = false end
        if hum then hum.PlatformStand = false end
        
        task.wait(0.1)
        
        if hrp then 
            hrp.CFrame = originalPos 
        end
    end)
end
